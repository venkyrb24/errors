  Step 1 — Create repos:                                                                                                                       
  git pull
  cd infrastructure                                                                                                                            
  terraform apply -var-file="cobol-conversion-pipeline.tfvars" -auto-approve                                                                   
  This creates cobol-pipeline-source-discovery, cobol-pipeline-structure-analyzer, etc. Lambda creation will fail (no images yet) — that's
  fine.                                                                                                                                        
                                                                                                                                               
  Step 2 — Build and push to new repos:
  $env:USERPROFILE = "C:\Users\$env:USERNAME"                                                                                                  
  aws ecr get-login-password --region us-east-1 --profile sba-prod | docker login --username AWS --password-stdin                              
  260851781623.dkr.ecr.us-east-1.amazonaws.com                                                                   
                                                                                                                                               
  $agents = @("source-discovery", "structure-analyzer", "data-mapper", "spec-synthesizer", "phase1-supervisor")
  foreach ($agent in $agents) {                                                                                                                
    $repo = "260851781623.dkr.ecr.us-east-1.amazonaws.com/cobol-pipeline-$agent`:latest"
    docker build --platform linux/amd64 --provenance=false --build-arg AGENT_NAME=$agent -t $repo -f Dockerfile.agent .                        
    docker push $repo                                                                                                                          
  }
