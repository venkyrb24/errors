 # 1. Create the state bucket                                                                                                                                        
  aws s3api create-bucket --bucket cafs-cobol-pipeline-tfstate --region us-east-1 --profile sba-prod                                                                  
                                                                                                                                                                      
  # 2. Enable versioning                                                                                                                                              
  aws s3api put-bucket-versioning --bucket cafs-cobol-pipeline-tfstate --versioning-configuration Status=Enabled --profile sba-prod                                   
                                                                                                                                                                      
  # 3. Enable encryption                                                                                                                                              
  aws s3api put-bucket-encryption --bucket cafs-cobol-pipeline-tfstate --server-side-encryption-configuration                                                         
  "{\"Rules\":[{\"ApplyServerSideEncryptionByDefault\":{\"SSEAlgorithm\":\"AES256\"}}]}" --profile sba-prod                                                           
   
  # 4. Block public access                                                                                                                                            
  aws s3api put-public-access-block --bucket cafs-cobol-pipeline-tfstate --public-access-block-configuration
  BlockPublicAcls=true,IgnorePublicAcls=true,BlockPublicPolicy=true,RestrictPublicBuckets=true --profile sba-prod                                                     
   
  # 5. Create DynamoDB lock table                                                                                                                                     
  aws dynamodb create-table --table-name cobol-pipeline-tflock --attribute-definitions AttributeName=LockID,AttributeType=S --key-schema
  AttributeName=LockID,KeyType=HASH --billing-mode PAY_PER_REQUEST --region us-east-1 --profile sba-prod                                                              
                                                            
  # 6. Pull latest, wipe old backend, re-init                                                                                                                         
  git pull                                                  
  Remove-Item -Recurse -Force .terraform                                                                                                                              
  terraform init -backend-config=backend.hcl -reconfigure   
                                                                                                                                                                      
  Note: step 3 uses escaped quotes — that's the PowerShell format. If you're on cmd.exe instead, use single quotes around the JSON like in the backend.hcl comments.
